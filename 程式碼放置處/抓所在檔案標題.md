```dataviewjs
let content = await dv.io.load(dv.current().file.path);  // 讀取目前檔案內容
let titles = [...content.matchAll(/^# (.+)$/gm)].map(m => m[1]);  // 抓出 # 開頭的大標題

let output = "```\n";  // 開始標記
output += "📌 主標題列表：\n";
for (let t of titles) {
  output += `# ${t}\n`;
}
output += "```";  // 結束標記

dv.paragraph(output);
```
